import 'dotenv/config'; import express from 'express'; import productosRouter from "./src/rutas/ProductosRouter.js"; import path from 'path'; import { fileURLToPath } from 'url'; import { dirname } from 'path'; const __filename = fileURLToPath(import.meta.url); const __dirname = dirname(__filename); const PORT = process.env.PORT || 3005; const app = express(); app.use( express.json() ); app.use(express.static(path.join(__dirname, 'public'))); app.use('/api/productos', productosRouter); app.listen(PORT, () => console.log(`http://localhost:${PORT}`));
